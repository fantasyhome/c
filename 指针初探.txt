//编译器的内部—用指针改变地址储存的值，
//达到子函数改变主函数中变量值的目的。
#include<stdio.h>
void interchange(int *,int *);
int main(void)
{
    int x=5,y=10;
    printf("originally x=%d and y=%d.\n",x,y);
    interchange(&x,&y);//把地址发给函数
    printf("Now x=%d and y=%d.\n",x,y);
    return 0;
}
void interchange(int * u,int * v)
{
    int temp;
    temp=*u;
    *u=*v;
    *v=temp;
}
